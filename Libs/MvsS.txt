local function SaveConfig(name)
    if not name or name == "" then
        return false, "Config name cannot be empty"
    end

    local configData = {
        PresetColor = PresetColor,
        CloseBind = CloseBind,
        Elements = {}
    }

    -- Save all tracked elements
    for id, element in pairs(ConfigElements) do
        if element.Type == "Toggle" then
            configData.Elements[id] = element.Value
        elseif element.Type == "Slider" then
            configData.Elements[id] = element.Value
        elseif element.Type == "Dropdown" then
            configData.Elements[id] = element.Value
        elseif element.Type == "Textbox" then
            configData.Elements[id] = element.Value
        elseif element.Type == "Bind" then
            configData.Elements[id] = element.Value
        end
    end

    Configs[name] = configData
    return true, "Config saved: " .. name
end

local function LoadConfig(name)
    if not Configs[name] then
        return false, "Config not found: " .. name
    end

    local config = Configs[name]
    PresetColor = config.PresetColor or PresetColor
    CloseBind = config.CloseBind or CloseBind

    -- Load all element values
    if config.Elements then
        for id, value in pairs(config.Elements) do
            if ConfigElements[id] then
                local element = ConfigElements[id]
                if element.LoadValue then
                    element.LoadValue(value)
                end
            end
        end
    end

    if lib.ChangePresetColor then
        lib:ChangePresetColor(PresetColor)
    end

    return true, "Config loaded: " .. name
end

local function MakeDraggable(topbarobject, object)
    local Dragging = nil
    local DragInput = nil
    local DragStart = nil
    local StartPosition = nil

    local function Update(input)
        local Delta = input.Position - DragStart
        local pos = UDim2.new(
            StartPosition.X.Scale,
            StartPosition.X.Offset + Delta.X,
            StartPosition.Y.Scale,
            StartPosition.Y.Offset + Delta.Y
        )
        object.Position = pos
    end

    topbarobject.InputBegan:Connect(
        function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                Dragging = true
                DragStart = input.Position
                StartPosition = object.Position

                input.Changed:Connect(
                    function()
                        if input.UserInputState == Enum.UserInputState.End then
                            Dragging = false
                        end
                    end
                )
            end
        end
    )

    topbarobject.InputChanged:Connect(
        function(input)
            if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
                DragInput = input
            end
        end
    )

    UserInputService.InputChanged:Connect(
        function(input)
            if input == DragInput and Dragging then
                Update(input)
            end
        end
    )
end

-- Reworked Smooth Fade Animations
local FADE_TIME = 0.4
local FADE_INFO = TweenInfo.new(FADE_TIME, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)

-- This function gathers all GUI elements we want to fade
local function getFadeElements(parent)
    local elements = {parent}
    for _, child in ipairs(parent:GetDescendants()) do
        table.insert(elements, child)
    end
    return elements
end

local function FadeInUI(main)
    main.Visible = true
    local elements = getFadeElements(main)

    for _, element in ipairs(elements) do
        local goal = {}
        local original = originalTransparencies[element]

        -- Restore transparency for elements that have these properties
        if element:IsA("TextLabel") or element:IsA("TextButton") or element:IsA("TextBox") then
            goal.TextTransparency = original or 0
        end
        if element:IsA("ImageLabel") or element:IsA("ImageButton") then
            goal.ImageTransparency = original or 0
        end
        if element:IsA("GuiObject") then
            -- Restore background, defaulting to 0 if not previously stored
            goal.BackgroundTransparency = (originalTransparencies[element] ~= nil) and originalTransparencies[element] or 0
        end

        if next(goal) then
            TweenService:Create(element, FADE_INFO, goal):Play()
        end
    end
end

local function FadeOutUI(main)
    local elements = getFadeElements(main)

    for _, element in ipairs(elements) do
        local goal = {}

        -- Store original transparency and set goal to fade out (become transparent)
        if element:IsA("TextLabel") or element:IsA("TextButton") or element:IsA("TextBox") then
            originalTransparencies[element] = element.TextTransparency
            goal.TextTransparency = 1
        end
        if element:IsA("ImageLabel") or element:IsA("ImageButton") then
            originalTransparencies[element] = element.ImageTransparency
            goal.ImageTransparency = 1
        end
        if element:IsA("GuiObject") then
            originalTransparencies[element] = element.BackgroundTransparency
            goal.BackgroundTransparency = 1
        end

        if next(goal) then
            TweenService:Create(element, FADE_INFO, goal):Play()
        end
    end

    -- Wait for the animation to finish before hiding the UI
    task.wait(FADE_TIME)
    main.Visible = false
end

    -- Config Tab for saving/loading
    function lib:ConfigTab(tab)
        local configTab = tab

        -- Config name input
        configTab:Textbox("Config Name", false, function(text)
            CurrentConfigTextbox = text
        end)

        -- Save config button
        configTab:Button("Save Config", function()
            local configName = CurrentConfigTextbox
            if configName and configName ~= "" then
                local success, msg = SaveConfig(configName)
                lib:Notification(success and "Success" or "Error", msg, "Okay")
            else
                lib:Notification("Error", "Please enter a config name first", "Okay")
            end
        end)

        -- Load config button
        configTab:Button("Load Config", function()
            local configName = CurrentConfigTextbox
            if configName and configName ~= "" then
                local success, msg = LoadConfig(configName)
                lib:Notification(success and "Success" or "Error", msg, "Okay")
            else
                lib:Notification("Error", "Please enter a config name first", "Okay")
            end
        end)

        -- List configs
        configTab:Button("List Configs", function()
            local configList = ""
            local count = 0
            for name, _ in pairs(Configs) do
                configList = configList .. name .. ", "
                count = count + 1
            end

            if count == 0 then
                configList = "No configs saved."
            else
                configList = string.sub(configList, 1, -3) -- Remove trailing comma and space
            end
            lib:Notification("Saved Configs", configList, "Okay")
        end)
    end

        function tabcontent:Bind(text, keypreset, callback)
            elementIdCounter = elementIdCounter + 1
            local elementId = "Bind_" .. tostring(os.time()) .. "_" .. tostring(elementIdCounter)
            local currentKey = keypreset
            
            local Bind = Instance.new("TextButton")
            local BindTitle = Instance.new("TextLabel")
            local BindText = Instance.new("TextLabel")

            Bind.Name = "Bind"
            Bind.Parent = Tab
            Bind.BackgroundColor3 = Color3.fromRGB(34, 34, 34)
            Bind.Size = UDim2.new(0, 363, 0, 42)
            Bind.AutoButtonColor = false
            Bind.Font = Enum.Font.SourceSans
            Bind.Text = ""

            local BindCorner = Instance.new("UICorner")
            BindCorner.CornerRadius = UDim.new(0, 6)
            BindCorner.Parent = Bind

            BindTitle.Name = "BindTitle"
            BindTitle.Parent = Bind
            BindTitle.BackgroundTransparency = 1.000
            BindTitle.Position = UDim2.new(0.0358, 0, 0, 0)
            BindTitle.Size = UDim2.new(0.5, 0, 1, 0)
            BindTitle.Font = Enum.Font.Gotham
            BindTitle.Text = text
            BindTitle.TextColor3 = Color3.fromRGB(255, 255, 255)
            BindTitle.TextSize = 14.000
            BindTitle.TextXAlignment = Enum.TextXAlignment.Left

            BindText.Name = "BindText"
            BindText.Parent = Bind
            BindText.BackgroundTransparency = 1.000
            BindText.Size = UDim2.new(1, -20, 1, 0)
            BindText.Font = Enum.Font.Gotham
            BindText.Text = currentKey.Name
            BindText.TextColor3 = Color3.fromRGB(255, 255, 255)
            BindText.TextSize = 14.000
            BindText.TextXAlignment = Enum.TextXAlignment.Right

            -- Save config element
            ConfigElements[elementId] = {
                Type = "Bind",
                Value = currentKey.Name,
                LoadValue = function(valueName)
                    local foundKey = Enum.KeyCode[valueName]
                    if foundKey then
                        currentKey = foundKey
                        BindText.Text = currentKey.Name
                    end
                end
            }

            Bind.MouseButton1Click:Connect(function()
                if isMobile then
                    lib:Notification("Keybinds", "Not supported on mobile.", "Okay")
                    return
                end
                
                BindText.Text = "..."
                local input, _ = UserInputService.InputBegan:Wait()
                if input.KeyCode ~= Enum.KeyCode.Unknown then
                    currentKey = input.KeyCode
                    BindText.Text = currentKey.Name
                    ConfigElements[elementId].Value = currentKey.Name
                    
                    if text:lower():match("toggle") or text:lower():match("ui") then
                        lib:SetToggleKey(currentKey)
                    end
                    pcall(callback, currentKey)
                else
                    BindText.Text = currentKey.Name -- Revert if unknown key
                end
            end)

            if not isMobile then
                UserInputService.InputBegan:Connect(function(input, gameProcessed)
                    if not gameProcessed and input.KeyCode == currentKey then
                        pcall(callback, currentKey)
                    end
                end)
            end
            
            Tab.CanvasSize = UDim2.new(0, 0, 0, TabLayout.AbsoluteContentSize.Y)
        end

        return tabcontent
    end
    return tabhold
end

return lib
